\section{Percentages}

\frame{
  \frametitle{Introduction to Percentages}

  \begin{itemize}
    \item cent means hundred
    \item percent means \colorbox{yellow}{per hundred}\ or \colorbox{yellow}{out of one hundred}. 
    \item So $50\%$ means 50 out of 100.
  \end{itemize}
  \pause 

  \begin{empheq}[box=\othermathbox]{align*}
    \text{To convert a number to a percentage: multiply by $100\%$}
  \end{empheq}
  \pause

  Questions:
  \begin{enumerate}
  \item What is $3/4$ as $\%$?
    \begin{equation*}
      A=0.75\%
      \quad 
      B = 30\%
      \quad 
      C = 7.5\%
      \quad 
      D = 75\%
      \quad
      \pause
      \answer{D}
    \end{equation*}
    \vspace*{-0.2in}

  \item What is $20\%$ of $30$?
    \begin{equation*}
      A=600
      \quad 
      B = 60
      \quad 
      C =6
      \quad 
      D = 0.6
      \quad
      \pause\answer{C}
    \end{equation*}
  \end{enumerate}
}
